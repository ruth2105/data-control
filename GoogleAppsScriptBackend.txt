function doOptions(e) {
  return ContentService.createTextOutput('')
    .setMimeType(ContentService.MimeType.TEXT)
    .setHeader('Access-Control-Allow-Origin', '*')
    .setHeader('Access-Control-Allow-Methods', 'GET,POST,OPTIONS')
    .setHeader('Access-Control-Allow-Headers', 'Content-Type,Accept');
}

function doPost(e) {
  try {
    const raw = e && e.postData && e.postData.contents ? e.postData.contents : '{}';
    const data = typeof raw === 'string' ? JSON.parse(raw) : raw;
    const action = data && data.action ? data.action : '';

    if (action === 'getCustomers') return getCustomers();
    if (action === 'updateCustomer') return updateCustomer(data);
    if (action === 'saveCustomers') return saveCustomers(data.data);
    if (action === 'getUsers') return getUsers();
    if (action === 'registerUser') return registerUser(data);
    if (action === 'approveUser') return approveUser(data);
    if (action === 'addUser') return addUser(data);
    if (action === 'updateUser') return updateUser(data);
    if (action === 'deleteUser') return deleteUser(data);
    if (action === 'saveUsers') return saveUsers(data.data);
    if (action === 'getVisitReports') return getVisitReports();
    if (action === 'saveVisitReports') return saveVisitReports(data.data);
    if (action === 'addVisitReport') return addVisitReport(data);
    if (action === 'updateVisitReport') return updateVisitReport(data);
    if (action === 'deleteVisitReport') return deleteVisitReport(data);
    if (action === 'authenticate') return authenticate(data);
    if (action === 'getPotentialCustomers') return getPotentialCustomers();
    if (action === 'savePotentialCustomers') return savePotentialCustomers(data.data);
    if (action === 'getFacilitySummaries') return getFacilitySummaries();
    if (action === 'saveFacilitySummaries') return saveFacilitySummaries(data.data);

    if (action === 'insert') return addVisitReport(data);
    if (action === 'update') return updateVisitReport(data);
    if (action === 'delete') return deleteVisitReport(data);

    if (!action && data && data.hospitalName) {
      return addVisitReport(data);
    }

    return error_('Unknown action: ' + action);
  } catch (err) {
    return error_(err && err.message ? err.message : String(err));
  }
}

function doGet(e) {
  try {
    const action = e && e.parameter && e.parameter.action ? e.parameter.action : '';

    if (action === 'getCustomers') return getCustomers();
    if (action === 'getUsers') return getUsers();
    if (action === 'getVisitReports') return getVisitReports();

    return getVisitReports();
  } catch (err) {
    return error_(err && err.message ? err.message : String(err));
  }
}

function updateCustomer(data) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    let sheet = ss.getSheetByName('Customers');
    
    if (!sheet) {
      return error_('Customers sheet not found');
    }
    
    const rows = sheet.getDataRange().getValues();
    const headers = rows[0];
    const customerId = data.id;
    
    // Find row with matching ID
    let rowIndex = -1;
    for (let i = 1; i < rows.length; i++) {
      if (String(rows[i][0]) === String(customerId)) {
        rowIndex = i + 1; // Sheet rows are 1-indexed
        break;
      }
    }
    
    if (rowIndex === -1) {
      // Customer not found, add new row
      data.createdAt = new Date().toISOString();
      data.updatedAt = new Date().toISOString();
      const row = headers.map(h => data[h] || '');
      sheet.appendRow(row);
    } else {
      // Update existing row
      data.updatedAt = new Date().toISOString();
      const row = headers.map(h => {
        return data[h] !== undefined ? data[h] : rows[rowIndex-1][headers.indexOf(h)];
      });
      sheet.getRange(rowIndex, 1, 1, row.length).setValues([row]);
    }
    
    return success_({message: 'Customer updated'});
    
  } catch(err) {
    return error_(err.message);
  }
}

function authenticate(data) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sheet = ss.getSheetByName('Users');
    
    if (!sheet) return error_('Users sheet not found');
    
    const rows = sheet.getDataRange().getValues();
    const headers = rows[0];
    const username = data.username;
    const password = data.password;
    
    // Find matching user
    for (let i = 1; i < rows.length; i++) {
      const row = rows[i];
      const user = {};
      headers.forEach((h, idx) => user[h] = row[idx]);
      
      if (user.username === username && user.password === password) {
        // Update last login
        const rowNum = i + 1;
        const lastLoginCol = headers.indexOf('lastLogin') + 1;
        sheet.getRange(rowNum, lastLoginCol).setValue(new Date().toISOString());
        
        // Return user info (without password)
        return success_({
          username: user.username,
          role: user.role,
          teamId: user.teamId,
          name: user.name,
          nameZh: user.nameZh
        });
      }
    }
    
    return error_('Invalid username or password');
    
  } catch(err) {
    return error_(err.message);
  }
}

function saveVisitReports(reports) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    let sheet = ss.getSheetByName('VisitReports');
    
    if (!sheet) {
      sheet = ss.getSheets()[0];
      sheet.setName('VisitReports');
    }
    
    // Clear and rewrite all data
    sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn()).clearContent();
    
    if (reports.length > 0) {
      const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
      const rows = reports.map(r => headers.map(h => {
        const v = r[h];
        return Array.isArray(v) ? v.join(', ') : (v || '');
      }));
      sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
    }
    
    return success_({message: 'Visit reports saved'});
    
  } catch(err) {
    return error_(err.message);
  }
}

function addVisitReport(data) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    let sheet = ss.getSheetByName('VisitReports');
    
    if (!sheet) {
      sheet = ss.getSheets()[0];
      sheet.setName('VisitReports');
    }
    
    // Ensure headers exist
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(VISIT_HEADERS);
    }
    
    // Add submission timestamp and ID if not present
    if (!data.submittedAt) data.submittedAt = new Date().toISOString();
    if (!data.id) data.id = 'V' + Date.now();
    
    const row = VISIT_HEADERS.map(h => {
      const v = data[h];
      return Array.isArray(v) ? v.join(', ') : (v || '');
    });
    sheet.appendRow(row);
    
    return success_({message: 'Visit report added', id: data.id});
    
  } catch(err) {
    return error_(err.message);
  }
}

function updateVisitReport(data) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sheet = ss.getSheetByName('VisitReports') || ss.getSheets()[0];
    
    const rows = sheet.getDataRange().getValues();
    const headers = rows[0];
    
    // Find report by ID
    let rowIndex = -1;
    for (let i = 1; i < rows.length; i++) {
      if (String(rows[i][0]) === String(data.id)) {
        rowIndex = i + 1;
        break;
      }
    }
    
    if (rowIndex === -1) return error_('Report not found');
    
    // Update row
    const row = VISIT_HEADERS.map(h => {
      const v = data[h];
      if (v !== undefined) {
        return Array.isArray(v) ? v.join(', ') : v;
      }
      return rows[rowIndex-1][VISIT_HEADERS.indexOf(h)] || '';
    });
    sheet.getRange(rowIndex, 1, 1, row.length).setValues([row]);
    
    return success_({message: 'Visit report updated'});
    
  } catch(err) {
    return error_(err.message);
  }
}

function deleteVisitReport(data) {

    const sheet = ss.getSheetByName('VisitReports') || ss.getSheets()[0];
    
    const rows = sheet.getDataRange().getValues();
    
    // Find and delete report
    for (let i = 1; i < rows.length; i++) {
      if (String(rows[i][0]) === String(data.id)) {

        return success_({message: 'Visit report deleted'});

  } catch(err) {
    return error_(err.message);
  }
}

function getFacilitySummaries() {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    let sheet = ss.getSheetByName('FacilitySummaries');
    
    if (!sheet) {
      sheet = ss.insertSheet('FacilitySummaries');
      const headers = ['hospitalName', 'summary'];
      sheet.appendRow(headers);
      return success_({});
    }
    
    const rows = sheet.getDataRange().getValues();
    if (rows.length <= 1) return success_({});
    
    const headers = rows[0];
    const data = rows.slice(1).map(row => {
      const obj = {};
      headers.forEach((h, i) => {
        obj[h] = row[i] === '' ? null : row[i];
      });
      return obj;
    });
    
    // Convert array format to object format
    const summaryObj = {};
    data.forEach(item => {
      if (item.hospitalName && item.summary) {
        summaryObj[item.hospitalName] = item.summary;
      }
    });
    
    return success_(summaryObj);
    
  } catch(err) {
    return error_(err.message);
  }
}

function saveFacilitySummaries(summaries) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    let sheet = ss.getSheetByName('FacilitySummaries');
    
    if (!sheet) {
      sheet = ss.insertSheet('FacilitySummaries');
      const headers = ['hospitalName', 'summary'];
      sheet.appendRow(headers);
    }
    
    // Convert object format to array format
    const summaryArray = Object.keys(summaries).map(hospitalName => ({
      hospitalName: hospitalName,
      summary: summaries[hospitalName]
    }));
    
    // Clear and rewrite all data
    sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn()).clearContent();
    
    if (summaryArray.length > 0) {
      const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
      const rows = summaryArray.map(s => headers.map(h => s[h] || ''));
      sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
    }
    
    return success_({message: 'Facility summaries saved'});
    
  } catch(err) {
    return error_(err.message);
  }
}

// ===================== HELPER FUNCTIONS =====================

function success_(data) {
  const output = ContentService.createTextOutput(JSON.stringify({
    status: 'success',
    data: data
  })).setMimeType(ContentService.MimeType.JSON);

  output.setHeader('Access-Control-Allow-Origin', '*');
  output.setHeader('Access-Control-Allow-Methods', 'GET,POST,OPTIONS');
  output.setHeader('Access-Control-Allow-Headers', 'Content-Type,Accept');
  return output;
}

function error_(message) {
  const output = ContentService.createTextOutput(JSON.stringify({
    status: 'error',
    message: message
  })).setMimeType(ContentService.MimeType.JSON);

  output.setHeader('Access-Control-Allow-Origin', '*');
  output.setHeader('Access-Control-Allow-Methods', 'GET,POST,OPTIONS');
  output.setHeader('Access-Control-Allow-Headers', 'Content-Type,Accept');
  return output;
}

// ===================== INITIALIZE FUNCTION =====================
// Run this once to set up all sheets

function initializeSheets() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  
  // Create Customers sheet
  let customersSheet = ss.getSheetByName('Customers');
  if (!customersSheet) {
    customersSheet = ss.insertSheet('Customers');
    const headers = ['id','name','nameZh','city','cityZh','model','contact','phone','email','doctor',
                     'lastVisit','teamId','potential','notes','address','specialty','type','createdAt','updatedAt'];
    customersSheet.appendRow(headers);
    Logger.log('✅ Customers sheet created');
  }
  
  // Create Users sheet
  let usersSheet = ss.getSheetByName('Users');
  if (!usersSheet) {
    usersSheet = ss.insertSheet('Users');
    const headers = ['username','password','role','teamId','name','nameZh','createdAt','lastLogin','status','phone','email','preferredTeam','registeredAt'];
    usersSheet.appendRow(headers);
    
    // Add default admin
    usersSheet.appendRow(['admin', 'admin123', 'admin', '', 'Administrator', '管理员', new Date().toISOString(), '', 'active', '', '', '', '']);
    
    Logger.log('✅ Users sheet created with admin user');
  }
  
  // Create or rename VisitReports sheet
  let reportsSheet = ss.getSheetByName('VisitReports');
  if (!reportsSheet) {
    reportsSheet = ss.getSheets()[0];
    reportsSheet.setName('VisitReports');
    Logger.log('✅ VisitReports sheet created/renamed');
  }
  
  // Create PotentialCustomers sheet
  let potentialsSheet = ss.getSheetByName('PotentialCustomers');
  if (!potentialsSheet) {
    potentialsSheet = ss.insertSheet('PotentialCustomers');
    const headers = ['name','address','contact','phone','patientLoad','hormoneAnalyzer','chemAnalyzer','electrolyteAnalyzer','testTubes','status','notes','salesRep','fromVisit','teamId'];
    potentialsSheet.appendRow(headers);
    Logger.log('✅ PotentialCustomers sheet created');
  }
  
  // Create FacilitySummaries sheet
  let summariesSheet = ss.getSheetByName('FacilitySummaries');
  if (!summariesSheet) {
    summariesSheet = ss.insertSheet('FacilitySummaries');
    const headers = ['hospitalName', 'summary'];
    summariesSheet.appendRow(headers);
    Logger.log('✅ FacilitySummaries sheet created');
  }
  
  Logger.log('✅ All sheets initialized successfully!');
}
